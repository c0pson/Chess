108    def __init__(self, color: str, board, position: tuple[int, int], notation_func) -> None:
109        super().__init__(color, board, position)
110        self.color: str = color # b | w
111        self.position: tuple[int, int] = position
112        self.board = board
113        self.load_image()
114        self.first_move: bool = True
115        self.moved_by_two: bool = False
116        self.can_en_passant: bool = False
117        self.move: int = 1 if self.color == 'b' else -1
118        self.notation_func: Callable = notation_func

120    def check_possible_moves(self, color: str, checking: bool=False) -> list[tuple[int, int]]:
121        if self.check_turn(color) and not checking:
122            return []
123        move = self.move
124        possible_moves: list[tuple[int, int]] = []
125        if self.position[0] in {0, 7}:
126            return possible_moves
127        forward_one = (self.position[0] + move, self.position[1])
128        forward_two = (self.position[0] + move * 2, self.position[1])
129        if not self.board.board[forward_one[0]][forward_one[1]].figure:
130            possible_moves.append(forward_one)
131            if self.first_move and not self.board.board[forward_two[0]][forward_two[1]].figure:
132                possible_moves.append(forward_two)
133        for offset in [-1, 1]:
134            if 0 <= self.position[1] + offset < 8:
135                capture_position = (self.position[0] + move, self.position[1] + offset)
136                target_square = self.board.board[capture_position[0]][capture_position[1]]
137                if target_square.figure and target_square.figure.color != self.color:
138                    possible_moves.append(capture_position)
139                adjacent_pawn_position = (self.position[0], self.position[1] + offset)
140                adjacent_pawn = self.board.board[adjacent_pawn_position[0]][adjacent_pawn_position[1]].figure
141                if isinstance(adjacent_pawn, Pawn) and adjacent_pawn.color != self.color and adjacent_pawn.moved_by_two:
142                    possible_moves.append((self.position[0] + move, self.position[1] + offset))
143                    self.can_en_passant = True
144        return possible_moves

182class Knight(Piece):
183    def __init__(self, color: str, board, position: tuple[int, int]) -> None:
184        super().__init__(color, board, position)
185        self.color: str = color
186        self.board = board
187        self.load_image()
188
189    def check_moves(self, exceptions: list[int]) -> list[tuple[int, int]]:
190        possible_moves: list[tuple[int, int]] = []
191        moves = [   (-2,-1), (-2, 1),
192                    (-1,-2), (-1, 2),
193                    ( 1,-2), ( 1, 2),
194                    ( 2,-1), ( 2, 1)   ] 
195        for i, move in enumerate(moves):
196            if i in exceptions:
197                continue
198            new_position = (self.position[0] + move[0], self.position[1] + move[1])
199            if 0 <= new_position[0] <= 7 and 0 <= new_position[1] <= 7:
200                target_square = self.board.board[new_position[0]][new_position[1]]
201                if not target_square.figure or target_square.figure.color != self.color:
202                    possible_moves.append(new_position)
203        return possible_moves
204
205    def check_possible_moves(self, color: str, checking: bool = False) -> list[tuple[int, int]]:
206        if self.check_turn(color) and not checking:
207            return []
208        special_cases = {
209            (1, 1): [0, 1, 2, 4],
210            (1, 6): [0, 1, 3, 5],
211            (6, 6): [3, 5, 6, 7],
212            (6, 1): [2, 4, 3, 5],
213            (0, 0): [0, 1, 2, 3, 4, 6],
214            (0, 1): [0, 1, 2, 3, 4],
215            (1, 0): [0, 1, 2, 4, 7]
216        }
217        if 2 <= self.position[0] <= 5 and self.position[1] in {1, 6} and self.position not in special_cases:
218            if self.position[1] == 1:
219                return self.check_moves([2, 4])
220            if self.position[1] == 6:
221                return self.check_moves([3, 5])
222        if 2 <= self.position[1] <= 5 and self.position[0] in {1, 6} and self.position not in special_cases:
223            if self.position[0] == 1:
224                return self.check_moves([0, 1])
225            if self.position[0] == 6:
226                return self.check_moves([6, 7])
227        if self.position in special_cases:
228            return self.check_moves(special_cases[self.position])
229        return self.check_moves([])

239        if self.check_turn(color) and not checking:

242        moves_vec = [

255                        possible_moves.append((x, y))
256                        break
257                    else:
258                        break
259                else:
260                    break

330class King(Piece):
331    def __init__(self, color: str, board, position: tuple[int, int]) -> None:
332        super().__init__(color, board, position)
333        self.color: str = color
334        self.board = board
335        self.load_image()
336        self.first_move: bool = True
337        self.can_castle: bool = False
338
339    def check_possible_moves(self, color: str, checking: bool = False) -> list[tuple[int, int]]:
340        if self.check_turn(color) and not checking:
341            return []
342        possible_moves: list[tuple[int, int]] = []
343        for i in range(max(0, self.position[0] - 1), min(8, self.position[0] + 2)):
344            for j in range(max(0, self.position[1] - 1), min(8, self.position[1] + 2)):
345                if not self.board.board[i][j].figure:
346                    possible_moves.append((i, j))
347                if self.board.board[i][j].figure and self.board.board[i][j].figure.color != self.color:
348                    possible_moves.append((i, j))
349        if self.first_move and not checking:
350            possible_moves.extend(self.get_castling_moves())
351        return possible_moves
352
353    def get_castling_moves(self) -> list[tuple[int, int]]:
354        castling_moves = []
355        row = self.position[0]
356        if self.can_castle_kingside():
357            castling_moves.append((row, 6))
358        if self.can_castle_queenside():
359            castling_moves.append((row, 2))
360        return castling_moves
361
362    def can_castle_kingside(self) -> bool:
363        row, col = self.position
364        if isinstance(self.board.board[row][7].figure, Rook) and self.board.board[row][7].figure.first_move:
365            for i in range(col + 1, 7):
366                if self.board.board[row][i].figure or self.board.is_under_attack((row, i), self.color):
367                    return False
368            if self.board.is_under_attack((row, 6), self.color) or self.board.is_under_attack((row, col), self.color):
369                return False
370            return True
371        return False
372
373    def can_castle_queenside(self) -> bool:
374        row, col = self.position
375        if isinstance(self.board.board[row][0].figure, Rook) and self.board.board[row][0].figure.first_move:
376            for i in range(col - 1, 0, -1):
377                if self.board.board[row][i].figure or self.board.is_under_attack((row, i), self.color):
378                    return False
379            if self.board.is_under_attack((row, 2), self.color) or self.board.is_under_attack((row, col), self.color):
380                return False
381            return True
382        return False
