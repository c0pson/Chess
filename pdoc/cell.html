187    def is_game_over(self) -> tuple[bool, bool]:
188        """Checks if checkmate occurred.
189
190        Returns:
191
192         - tuple[bool, bool]: 1st tuple element is game_over and 2nd is in check both True or False.
193        """
194        in_check = False
195        for row in self.board:
196            for cell in row:
197                if cell.figure and cell.figure.color == self.current_turn:
198                    possible_moves = cell.figure.check_possible_moves(self.current_turn)
199                    for move in possible_moves:
200                        if not self.check_check(cell.figure.position, move):
201                            return False, False
202        for row in self.board:

204                if isinstance(cell.figure, piece.King) and cell.figure.color == self.current_turn:
205                    if self.check_check(cell.figure.position, cell.figure.position):
206                        in_check = True
207                        break
208        return True, in_check

211        """Handles actions after clicking on a specific cell.

237    def check_check(self, move_from: tuple[int, int], move_to: tuple[int, int]) -> bool:
238        """Checks if King is in a check.
239
240        Args:
241
242         - move_from (tuple[int, int]): Starting position.
243         - move_to (tuple[int, int]): Desired position.
244
245        Returns:
246
247         - bool: _description_
248        """
249        original_from_figure: piece.Piece | None = self.board[move_from[0]][move_from[1]].figure
250        original_to_figure: piece.Piece | None = self.board[move_to[0]][move_to[1]].figure
251        self.board[move_to[0]][move_to[1]].figure = original_from_figure
252        self.board[move_from[0]][move_from[1]].figure = None
253        king_position = None
254        if isinstance(original_from_figure, piece.King):
255            king_position = move_to
256        else:
257            for row in self.board:
258                for cell in row:
259                    if isinstance(cell.figure, piece.King) and cell.figure.color == self.current_turn:
260                        king_position = cell.figure.position
261                        break
262                if king_position:
263                    break
264        is_in_check = False
265        for row in self.board:
266            for cell in row:
267                if cell.figure and cell.figure.color != self.current_turn:
268                    possible_moves = cell.figure.check_possible_moves(cell.figure.color)
269                    if king_position in possible_moves:
270                        is_in_check = True
271                        break
272            if is_in_check:
273                break
274        self.board[move_from[0]][move_from[1]].figure = original_from_figure
275        self.board[move_to[0]][move_to[1]].figure = original_to_figure
276        return is_in_check

296    def handle_move(self, position: tuple[int, int]) -> None:
297        """Function handles moving pieces on the board.
298
299        Args:
300
301         - position (tuple[int, int]): Position of the figure.
302        """
303        if self.clicked_figure and self.previous_coords:
304            row, col = position
305            cell = self.board[row][col]
306            capture = bool(cell.figure)
307            self.capture = bool(cell.figure)
308            promotion = False
309            if cell in self.highlighted and self.previous_coords != position:
310                castle = False
311                if not self.check_check(self.previous_coords, position):
312                    if isinstance(self.clicked_figure, piece.Pawn) and self.clicked_figure.can_en_passant and col != self.previous_coords[1] and not cell.figure:
313                        self.board[row - self.clicked_figure.move][col].figure = None
314                        self.board[row - self.clicked_figure.move][col].update()
315                    if isinstance(self.clicked_figure, piece.King):
316                        if abs(col - self.previous_coords[1]) == 2:
317                            if col == 6:
318                                self.board[row][5].figure = self.board[row][7].figure
319                                self.board[row][7].figure = None
320                                self.board[row][5].figure.position = (row, 5) # type: ignore
321                                self.board[row][5].update()
322                                self.board[row][7].update()
323                                self.moves_record.record_move(self.clicked_figure, castle="kingside")
324                                castle = True
325                            elif col == 2:
326                                self.board[row][3].figure = self.board[row][0].figure
327                                self.board[row][0].figure = None
328                                self.board[row][3].figure.position = (row, 3) # type: ignore
329                                self.board[row][3].update()
330                                self.board[row][0].update()
331                                self.moves_record.record_move(self.clicked_figure, castle="queenside")
332                                castle =True
333                    cell.figure = self.clicked_figure
334                    cell.figure.position = position
335                    cell.update()
336                    self.board[self.previous_coords[0]][self.previous_coords[1]].figure = None
337                    self.board[self.previous_coords[0]][self.previous_coords[1]].update()
338                    if isinstance(cell.figure, piece.Pawn):
339                        if cell.figure.first_move and abs(self.previous_coords[0] - row) == 2:
340                            cell.figure.moved_by_two = True
341                        else:
342                            cell.figure.moved_by_two = False
343                        if cell.figure.promote():
344                            promotion = True
345                    self.reset_en_passant_flags(cell.figure.color)
346                    if cell.figure.first_move:
347                        cell.figure.first_move = False
348                    self.current_turn = 'b' if self.current_turn == 'w' else 'w'
349                    check = self.is_under_attack(self.get_king_position(self.current_turn), self.current_turn)
350                    game_over, in_check = self.is_game_over()
351                    if game_over:
352                        if in_check:
353                            self.display_message(f'Checkmate  {"White wins!" if self.current_turn == "b" else "Black wins!"}', 9)
354                            if not promotion:
355                                self.moves_record.record_move(self.clicked_figure, capture=capture, previous_coords=self.previous_coords, check=check, checkmate=game_over and in_check)
356                        else:
357                            self.display_message('Stalemate', 9)
358                            if not promotion:
359                                self.moves_record.record_move(self.clicked_figure, capture=capture, previous_coords=self.previous_coords, check=check, checkmate=game_over and in_check)
360                    elif not castle and not promotion:
361                        self.moves_record.record_move(self.clicked_figure, capture=capture, previous_coords=self.previous_coords, check=check, checkmate=game_over and in_check)
362            if not promotion:
363                self.clicked_figure = None
364                self.previous_coords = None
365        self.remove_highlights()
